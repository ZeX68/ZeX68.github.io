  • 对于折光率和质量百分含量，我们要进行拟合作图。我们选择2-4次的函数拟合。

    from scipy import optimize
def fun_2(x, a, b, c):
    return a*x**2 +b*x +c
 
def fun_3(x, a, b, c,d):
    return a*x**3 +b*x**2 +c*x+d
 
def fun_4(x, a,b,c,d,e):
    return a*x**4 +b*x**3 +c*x**2+d*x+e

parameters_2 = optimize.curve_fit(fun_2, reflac, concentrate)[0]
parameters_3 = optimize.curve_fit(fun_3, reflac, concentrate)[0]
parameters_4 = optimize.curve_fit(fun_4, reflac, concentrate)[0]

  • 定义函数。 

    def get_R_square(x, y, params,func):
    y_mean = np.mean(y)
    y_predict = [func(e, *params) for e in x]
    TSS = [(x-y_mean)**2 for x in y]
    RSS = [(y_predict[i] - y[i])**2 for i in range(len(y))]
    return 1 - sum(RSS) / sum(TSS)
 
r2 = get_R_square(reflac, concentrate, parameters_2, fun_2)
r3 = get_R_square(reflac, concentrate, parameters_3, fun_3)
r4 = get_R_square(reflac, concentrate, parameters_4, fun_4)
 
r_2 = [r2,r3,r4]
 
fig, ax = plt.subplots(1,3 ,figsize=(10,6),dpi=300)

第二步我们需要作出二元相图，此时选择二次曲线作为工作曲线。

def work_curve(x):
    y_pred = 0
    for i in range(3):
        y_pred += parameters_2[i] * pow(x, 2-i)
    return y_pred
